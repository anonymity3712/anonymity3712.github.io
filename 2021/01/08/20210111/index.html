工作中的笔记整理---利用爬虫把html文件批量写入数据库

2021-01-08
工作笔记-python

要把几万条html文件按一定标签写入数据库

#coding:utf-8
import os
import re
import sys
import MySQLdb
from lxml import etree

path = 'aa'
# aa为html存放的文件夹
pattern0 = u' '
pattern1 = re.compile(r'(.*)(?=-)')
for docs in os.listdir(path):
    if os.path.isdir('aa/' + docs):
        print "目录跳过"
        continue
    #打开文件，提取内容javascript:;
    doc = open('aa/' + docs,'r')
    html = doc.read()
    doc.close()

    #提取信息

    xml = etree.HTML(html)

    if(xml.xpath("//title")):
                title=xml.xpath("//title")[0].text.replace(pattern0,'')
        else:
                continue

    #    author = xml.xpath("//a[@class='author name ng-binding']")[0].text.replace('    ','').replace(' ','').replace('\n','')

    pic = xml.xpath("//pic[@class='ng-pic']")[0].text

    doc = ("tjg/a/") + docs
    #doc[0]
    print title,pic,doc


    try:
        conn = MySQLdb.connect(host='localhost',port=3306,user='name',passwd='passwd',db='db',charset='utf8')
        cur = conn.cursor()
        reload(sys)
        sys.setdefaultencoding('utf-8')
        tmp = (title,pic,doc)
        cur.execute("INSERT INTO `tjg`(`title`,`pic`,`doc`) VALUES(%s,%s,%s)", tmp)
        conn.commit()
        cur.close()
        conn.close()
    except MySQLdb.Error,e:
            print "Mysql Error %d: %s" % (e.args[0], e.args[1])
